; Tone settings

Mode:      2     ; Measurement mode
                 ;   0 = Horizontal speed
                 ;   1 = Vertical speed
                 ;   2 = Glide ratio
                 ;   3 = Inverse glide ratio
                 ;   4 = Total speed
                 ;   11 = Dive angle
Min:       300   ; Lowest pitch value
                 ;   cm/s        in Mode 0, 1, or 4
                 ;   ratio * 100 in Mode 2 or 3
                 ;   degrees     in Mode 11
Max:       400   ; Highest pitch value
                 ;   cm/s        in Mode 0, 1, or 4
                 ;   ratio * 100 in Mode 2 or 3
                 ;   degrees     in Mode 11

; Rate settings

Mode_2:    9     ; Determines tone rate
                 ;   0 = Horizontal speed
                 ;   1 = Vertical speed
                 ;   2 = Glide ratio
                 ;   3 = Inverse glide ratio
                 ;   4 = Total speed
                 ;   8 = Magnitude of Value 1
                 ;   9 = Change in Value 1
                 ;   11 = Dive angle
Min_Val_2: 300   ; Lowest rate value
                 ;   cm/s          when Mode 2 = 0, 1, or 4
                 ;   ratio * 100   when Mode 2 = 2 or 3
                 ;   percent * 100 when Mode 2 = 9
                 ;   degrees       when Mode 2 = 11
Max_Val_2: 1500  ; Highest rate value
                 ;   cm/s          when Mode 2 = 0, 1, or 4
                 ;   ratio * 100   when Mode 2 = 2 or 3
                 ;   percent * 100 when Mode 2 = 9
                 ;   degrees       when Mode 2 = 11

; Speech settings

Sp_Rate:   0     ; Speech rate (s)
                 ;   0 = No speech
Sp_Volume: 8     ; 0 (min) to 8 (max)

Sp_Mode:   0     ; Speech mode
                 ;   0 = Horizontal speed
                 ;   1 = Vertical speed
                 ;   2 = Glide ratio
                 ;   3 = Inverse glide ratio
                 ;   4 = Total speed
                 ;   5 = Altitude above DZ_Elev
                 ;   11 = Dive angle
Sp_Units:  0     ; Speech units
                 ;   0 = km/h or m
                 ;   1 = mph or feet
Sp_Dec:    0     ; Speech precision
                 ;   Altitude step in Mode 5
                 ;   Decimal places in all other Modes

; Thresholds

; Disable tones
V_Thresh:  11111 ; Minimum vertical speed for tone (cm/s)
H_Thresh:  11111 ; Minimum horizontal speed for tone (cm/s)

; Acro settings

Acro_Win:           2286 ; Acro window, meters below exit
Exit_Min_Alt:       2286 ; Do not detect an exit if below this altitude in meters
                         ; (negative to disable exit detection)

; Initialization

Init_Mode: 2     ; When the FlySight is powered on
                 ;   0 = Do nothing
                 ;   1 = Test speech mode
                 ;   2 = Play file
Init_File: acro  ; File to be played
